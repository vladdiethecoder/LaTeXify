\begin{theorem}
Let $f$ be continuous on $[a,b]$ and differentiable on $(a,b)$. There exists
$\xi \in (a,b)$ such that
\[
  f(b) - f(a) = f'(\xi)(b-a).
\]
\end{theorem}
