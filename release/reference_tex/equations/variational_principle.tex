\begin{align}
  \delta S &= \delta \int_{t_0}^{t_1} L(q, \dot{q}, t)\, \mathrm{d}t = 0, \\
  \frac{\mathrm{d}}{\mathrm{d}t} \left( \frac{\partial L}{\partial \dot{q}_i} \right) - \frac{\partial L}{\partial q_i} &= 0, \qquad i = 1,\ldots,n.
\end{align}

\begin{align}
  \mathcal{H}(q, p, t) &= \sum_{i=1}^{n} p_i \dot{q}_i - L(q, \dot{q}, t), \\
  \dot{q}_i &= \frac{\partial \mathcal{H}}{\partial p_i}, \qquad
  \dot{p}_i = -\frac{\partial \mathcal{H}}{\partial q_i},
\end{align}

which establish the canonical Hamilton equations for conservative systems.
